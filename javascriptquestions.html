JavaScript Concepts

Var, Let, and Const

let x = 10;
const y = 20;

Hoisting

console.log(a); // undefined
var a = 5;

The Temporal Dead Zone

console.log(b); // ReferenceError
let b = 10;

Closures

function outer() {
  let c = 5;
  function inner() {
    console.log(c);
  }
  return inner;
}
const innerFunction = outer();
innerFunction(); // Outputs 5

Object and its methods

const person = {
  name: 'John',
  age: 30,
  greet: function() {
    console.log(`Hello, I'm ${this.name}`);
  }
};
person.greet(); // Outputs: Hello, I'm John

Callback Function

function doSomething(callback) {
  console.log("Doing something...");
  callback();
}
function callbackFunction() {
  console.log("Callback executed.");
}
doSomething(callbackFunction);

JavaScript Map

const myMap = new Map();
myMap.set('key1', 'value1');
myMap.set('key2', 'value2');
console.log(myMap.get('key1')); // Outputs: value1

Reduce Method

const numbers = [1, 2, 3, 4, 5];
const sum = numbers.reduce((accumulator, currentValue) => accumulator + currentValue, 0);
console.log(sum); // Outputs: 15

Unique Values - Set

const uniqueNumbers = new Set([1, 2, 2, 3, 4, 4]);
console.log(uniqueNumbers); // Outputs: Set { 1, 2, 3, 4 }

Destructuring

const person = { name: 'Alice', age: 25 };
const { name, age } = person;
console.log(name, age); // Outputs: Alice 25

Async/Await

async function fetchData() {
  const response = await fetch('https://api.example.com/data');
  const data = await response.json();
  console.log(data);
}
fetchData();

Promises

const fetchData = () => {
  return new Promise((resolve, reject) => {
    fetch('https://api.example.com/data')
      .then(response => response.json())
      .then(data => resolve(data))
      .catch(error => reject(error));
  });
};
fetchData()
  .then(data => console.log(data))
  .catch(error => console.error(error));
